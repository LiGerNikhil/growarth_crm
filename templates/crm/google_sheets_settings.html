JavaScript Integration Examples
Loan Form Integration
// Google Sheets Web App URL
const scriptURL = "https://script.google.com/macros/s/AKfycby-pJuT6CjTbSViGQd3qgOYzKQe-KPH0IQJ9UZCrujWCLXnnUeqWqz6sXkHOQ28tbiz-A/exec";

// Form submission
fetch(scriptURL, {
  method: 'POST',
  body: JSON.stringify(formData),
  headers: {
    'Content-Type': 'application/json'
  }
})
.then(response => response.json())
.then(data => {
  // Also send to local CRM
  fetch('/crm/webhook/loan-lead/', {
    method: 'POST',
    headers: {
      'Content-Type': 'application/json',
      'X-CSRFToken': getCookie('csrftoken')
    },
    body: JSON.stringify(formData)
  });
});
